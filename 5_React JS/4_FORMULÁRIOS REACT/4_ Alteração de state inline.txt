1_ Quando temos vários inputs podemos realizar a manipulação de forma mais simples;

2_ Basicamente criamos uma função inline no onChange;

3_ Ela vai alterar o valor do state com o método "set", da mesma forma que a função isolada;

4_ Vamos ver isto da prática;


**************************Myform.js***************************

import "./MyForm.css";
import {useState} from 'react';

const MyForm = () => {
  const [email, setEmail] = useState();
  
  console.log(email);

  return (
    <div>
        <form>
            <div>
                <label>
                  <span>Email:</span>
                  //Mudando state através de uma função anônima que recebe o evento.
                  <input type="email" name="email" placeholder="Digite seu email" onChange={(e) => {setEmail(e.target.value)}}/>
                </label>
            </div>
            <input type="submit" value="enviar"/>
        </form>
    </div>
  )
}

export default MyForm
